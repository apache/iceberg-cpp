182 Result<R> Predicate(const std::shared_ptr<BoundPredicate>& pred) override {
183 ICEBERG_DCHECK(pred != nullptr, "BoundPredicate cannot be null");
184
185 switch (pred->kind()) {
186 case BoundPredicate::Kind::kUnary: {
187 switch (pred->op()) {
188 case Expression::Operation::kIsNull:
189 return IsNull(pred->term());
190 case Expression::Operation::kNotNull:
191 return NotNull(pred->term());
192 case Expression::Operation::kIsNan:
193 return IsNaN(pred->term());
194 case Expression::Operation::kNotNan:
195 return NotNaN(pred->term());
196 default:
197 return InvalidExpression("Invalid operation for BoundUnaryPredicate: {}",
198 ToString(pred->op()));
199 }
200 }
201 case BoundPredicate::Kind::kLiteral: {
202 const auto& literal_pred =
203 internal::checked_cast<const BoundLiteralPredicate&>(*pred);
204 switch (pred->op()) {
205 case Expression::Operation::kLt:
206 return Lt(pred->term(), literal_pred.literal());
207 case Expression::Operation::kLtEq:
208 return LtEq(pred->term(), literal_pred.literal());
209 case Expression::Operation::kGt:
210 return Gt(pred->term(), literal_pred.literal());
211 case Expression::Operation::kGtEq:
212 return GtEq(pred->term(), literal_pred.literal());
213 case Expression::Operation::kEq:
214 return Eq(pred->term(), literal_pred.literal());
215 case Expression::Operation::kNotEq:
216 return NotEq(pred->term(), literal_pred.literal());
217 case Expression::Operation::kStartsWith:
218 return StartsWith(pred->term(), literal_pred.literal());
219 case Expression::Operation::kNotStartsWith:
220 return NotStartsWith(pred->term(), literal_pred.literal());
221 default:
222 return InvalidExpression("Invalid operation for BoundLiteralPredicate: {}",
223 ToString(pred->op()));
224 }
225 }
226 case BoundPredicate::Kind::kSet: {
227 const auto& set_pred = internal::checked_cast<const BoundSetPredicate&>(*pred);
228 switch (pred->op()) {
229 case Expression::Operation::kIn:
230 return In(pred->term(), set_pred.literal_set());
231 case Expression::Operation::kNotIn:
232 return NotIn(pred->term(), set_pred.literal_set());
233 default:
234 return InvalidExpression("Invalid operation for BoundSetPredicate: {}",
235 ToString(pred->op()));
236 }
237 }
238 }
239
240 return InvalidExpression("Unsupported bound predicate: {}", pred->ToString());
241 }
242
248 Result<R> Predicate(const std::shared_ptr<UnboundPredicate>& pred) final {
249 ICEBERG_DCHECK(pred != nullptr, "UnboundPredicate cannot be null");
250 return NotSupported("Not a bound predicate: {}", pred->ToString());
251 }

265template <typename R, typename V>
266 requires std::derived_from<V, ExpressionVisitor<R>>
267Result<R> Visit(const std::shared_ptr<Expression>& expr, V& visitor) {
268 ICEBERG_DCHECK(expr != nullptr, "Expression cannot be null");
269
270 if (expr->is_bound_predicate()) {
271 return visitor.Predicate(std::dynamic_pointer_cast<BoundPredicate>(expr));
272 }
273
274 if (expr->is_unbound_predicate()) {
275 return visitor.Predicate(std::dynamic_pointer_cast<UnboundPredicate>(expr));
276 }
277
278 // TODO(gangwu): handle aggregate expression
279
280 switch (expr->op()) {
281 case Expression::Operation::kTrue:
282 return visitor.AlwaysTrue();
283 case Expression::Operation::kFalse:
284 return visitor.AlwaysFalse();
285 case Expression::Operation::kNot: {
286 const auto& not_expr = internal::checked_pointer_cast<Not>(expr);
287 ICEBERG_ASSIGN_OR_RAISE(auto child_result,
288 (Visit<R, V>(not_expr->child(), visitor)));
289 return visitor.Not(std::move(child_result));
290 }
291 case Expression::Operation::kAnd: {
292 const auto& and_expr = internal::checked_pointer_cast<And>(expr);
293 ICEBERG_ASSIGN_OR_RAISE(auto left_result, (Visit<R, V>(and_expr->left(), visitor)));
294 ICEBERG_ASSIGN_OR_RAISE(auto right_result,
295 (Visit<R, V>(and_expr->right(), visitor)));
296 return visitor.And(std::move(left_result), std::move(right_result));
297 }
298 case Expression::Operation::kOr: {
299 const auto& or_expr = internal::checked_pointer_cast<Or>(expr);
300 ICEBERG_ASSIGN_OR_RAISE(auto left_result, (Visit<R, V>(or_expr->left(), visitor)));
301 ICEBERG_ASSIGN_OR_RAISE(auto right_result,
302 (Visit<R, V>(or_expr->right(), visitor)));
303 return visitor.Or(std::move(left_result), std::move(right_result));
304 }
305 default:
306 return InvalidExpression("Unknown expression operation: {}", expr->ToString());
307 }
308}
