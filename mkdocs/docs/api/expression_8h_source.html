137class ICEBERG_EXPORT And : public Expression {
138 public:
143 static Result<std::unique_ptr<And>> Make(std::shared_ptr<Expression> left,
144 std::shared_ptr<Expression> right);
145
155 template <typename... Args>
156 static Result<std::shared_ptr<Expression>> MakeFolded(std::shared_ptr<Expression> left,
157 std::shared_ptr<Expression> right,
158 Args&&... args)
159 requires std::conjunction_v<std::is_same<Args, std::shared_ptr<Expression>>...>
160 {
161 if constexpr (sizeof...(args) == 0) {
162 if (left->op() == Expression::Operation::kFalse ||
163 right->op() == Expression::Operation::kFalse) {
164 return False::Instance();
165 }
166
167 if (left->op() == Expression::Operation::kTrue) {
168 return right;
169 }
170
171 if (right->op() == Expression::Operation::kTrue) {
172 return left;
173 }
174
175 return And::Make(std::move(left), std::move(right));
176 } else {
177 ICEBERG_ASSIGN_OR_THROW(auto and_expr,
178 And::Make(std::move(left), std::move(right)));
179
180 return And::MakeFolded(std::move(and_expr), std::forward<Args>(args)...);
181 }
182 }
183
187 const std::shared_ptr<Expression>& left() const { return left_; }
188
192 const std::shared_ptr<Expression>& right() const { return right_; }
193
194 Operation op() const override { return Operation::kAnd; }
195
196 std::string ToString() const override;
197
198 Result<std::shared_ptr<Expression>> Negate() const override;
199
200 bool Equals(const Expression& other) const override;
201
202 private:
203 And(std::shared_ptr<Expression> left, std::shared_ptr<Expression> right);
204
205 std::shared_ptr<Expression> left_;
206 std::shared_ptr<Expression> right_;
207};
